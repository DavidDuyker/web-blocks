üíª Code Examples

Direct Web Component Usage

// Create elements directly const button = document.createElement('wc-button'); button.textContent = 'Click me'; button.setAttribute('href', 'https://example.com'); document.body.appendChild(button); const input = document.createElement('wc-input'); input.setAttribute('label', 'Email'); input.setAttribute('type', 'email'); input.setAttribute('placeholder', 'Enter your email'); document.body.appendChild(input);

Using JavaScript Classes

// Using the Button class const button = new Button({ label: 'Click me', href: '/some-page', onClick: () => console.log('clicked') }); document.body.appendChild(button.render()); // Using the Input class const input = new Input({ label: 'Email', type: 'email', placeholder: 'Enter your email', onChange: (e) => console.log('Value changed:', e.detail.value) }); document.body.appendChild(input.render()); // Using the Link class const link = new Link({ label: 'Visit our site', href: 'https://example.com', variant: 'crosshair' }); document.body.appendChild(link.render());
