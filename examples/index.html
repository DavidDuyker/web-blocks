üöÄ How to Use

1. Include the Components

<!-- Include design tokens and reset styles --> <link rel="stylesheet" href="./src/styles/tokens.css"> <link rel="stylesheet" href="./src/styles/reset.css"> <!-- Include the component you want to use --> <script src="./src/components/button/button.html"></script> <script src="./src/components/card/card.html"></script> <script src="./src/components/input/input.html"></script> <script src="./src/components/badge/badge.html"></script> <script src="./src/components/modal/modal.html"></script>

2. Use in Your HTML

<!-- No JavaScript required! --> <wc-button variant="primary">Click me</wc-button> <wc-card variant="elevated"> <div slot="body">Card content</div> </wc-card>

3. Listen to Events

<script> // Listen to component events document.addEventListener('wc-button-click', (e) => { console.log('Button clicked!', e.detail); }); document.addEventListener('wc-modal-open', (e) => { console.log('Modal opened!', e.detail); }); // Programmatic control const modal = document.getElementById('my-modal'); modal.setAttribute('open', ''); // Open modal modal.removeAttribute('open'); // Close modal </script>
